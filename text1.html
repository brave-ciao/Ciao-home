def computeCost(X, y, theta):
    inner = np.power(((X * theta.T) -  y), 2)
    return np.sum(inner) / (2 * len(X))     #a = np.sum([[0,4],[2,6]]) a:12

def gradientDescent(X, y, theta, alpha, epoch):
    """reuturn theta, cost"""
    
    temp = np.matrix(np.zeros(theta.shape))  # zeros初始化一个 θ 临时矩阵,括号内为几行几列，这里是(1, 2)
    parameters = int(theta.flatten().shape[1])  # 参数 θ的数量“flatten是numpy.ndarray.flatten的一个函数,即返回一个一维数组。flatten只能适用于numpy对象,即array或者mat,普通的list列表不适用!。 a.flatten():a是个数组,a.flatten()就是把a降到一维...”
    cost = np.zeros(epoch)  # 初始化一个ndarray，包含每次epoch的cost
    m = X.shape[0]  # 样本数量m
    
    for i in range(epoch):
        # 利用向量化一步求解
        temp =theta - (alpha / m) * (X * theta.T - y).T * X   #注意是点乘
        
# 以下是不用Vectorization求解梯度下降
#         error = (X * theta.T) - y  # (97, 1)
        
#         for j in range(parameters):
#             term = np.multiply(error, X[:,j])  # (97, 1)
#             temp[0,j] = theta[0,j] - ((alpha / m) * np.sum(term))  # (1,1)
            
        theta = temp
        cost[i] = computeCost(X, y, theta)
    return theta, cost

# 正规方程θ=(XT X)−1 XT y 
def normalEqn(X, y):
    theta = np.linalg.inv(X.T@X)@X.T@y#X.T@X等价于X.T.dot(X) 这里是矩阵相乘
    return theta
