def gradientDescent(X, y, theta, alpha, epoch):
    """reuturn theta, cost"""
    
    temp = np.matrix(np.zeros(theta.shape))  # zeros  ,(1, 2)
    parameters = int(theta.flatten().shape[1])  #  flattennumpy.ndarray.flatten,flattennumpy,arraymat,list! a.flatten():a,a.flatten()a...
    cost = np.zeros(epoch)  # ndarrayepochcost
    m = X.shape[0]  # m
    
    for i in range(epoch):
        # 
        temp =theta - (alpha / m) * (X * theta.T - y).T * X   #
        
# Vectorization
#         error = (X * theta.T) - y  # (97, 1)
        
#         for j in range(parameters):
#             term = np.multiply(error, X[:,j])  # (97, 1)
#             temp[0,j] = theta[0,j] - ((alpha / m) * np.sum(term))  # (1,1)
            
        theta = temp
        cost[i] = computeCost(X, y, theta)
    return theta, cost
